

## 3212 统计X 和 Y频数相等的子矩阵数量

> s[i][j] = s[i - 1][j] + s[i][j - 1] - s[i - 1][j - 1] + grid[i - 1][j - 1]
> > 剪掉 s[i - 1][j - 1]  是因为重复算了一次
```cpp
class Solution {
public:
    int numberOfSubmatrices(vector<vector<char>>& grid) {
        int rows = grid.size();
        int cols = grid[0].size();

        vector<vector<int>> preX(rows + 1, vector<int>(cols + 1, 0));
        vector<vector<int>> preY(rows + 1, vector<int>(cols + 1, 0));
        int ans = 0;
        
        for(int i = 1; i <= rows; i++){
            for(int j = 1; j <= cols; j++){
                preX[i][j] = preX[i - 1][j] + preX[i][j - 1] - preX[i - 1][j - 1] + (grid[i - 1][j - 1] == 'X' ? 1 : 0);

                preY[i][j] = preY[i - 1][j] + preY[i][j - 1] - preY[i - 1][j - 1] + (grid[i - 1][j - 1] == 'Y' ? 1 : 0);

                if(preX[i][j] > 0 && preX[i][j]== preY[i][j]){
                    ans++;
                }
            }
        }
        return ans;
    }
};

```

## 3642 垂直翻转子矩阵

> 垂直翻转 = 沿着水平轴做镜像，实现方式就是交换对称的行

```cpp
vector<vector<int>> reverseSubmatrix(vector<vector<int>>& grid, int x, int y, int k){
    int l = x, r = x + k - 1;
    while(l < r){
        for(int j = y; j < y + k; j++){
            swap(grid[l][j], grid[r][j]);
        }
        l++;
        r--;
    }
    return grid;
}
