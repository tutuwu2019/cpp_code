

## 3212 统计X 和 Y频数相等的子矩阵数量

> s[i][j] = s[i - 1][j] + s[i][j - 1] - s[i - 1][j - 1] + grid[i - 1][j - 1]
> > 剪掉 s[i - 1][j - 1]  是因为重复算了一次
```cpp
class Solution {
public:
    int numberOfSubmatrices(vector<vector<char>>& grid) {
        int rows = grid.size();
        int cols = grid[0].size();

        vector<vector<int>> preX(rows + 1, vector<int>(cols + 1, 0));
        vector<vector<int>> preY(rows + 1, vector<int>(cols + 1, 0));
        int ans = 0;
        
        for(int i = 1; i <= rows; i++){
            for(int j = 1; j <= cols; j++){
                preX[i][j] = preX[i - 1][j] + preX[i][j - 1] - preX[i - 1][j - 1] + (grid[i - 1][j - 1] == 'X' ? 1 : 0);

                preY[i][j] = preY[i - 1][j] + preY[i][j - 1] - preY[i - 1][j - 1] + (grid[i - 1][j - 1] == 'Y' ? 1 : 0);

                if(preX[i][j] > 0 && preX[i][j]== preY[i][j]){
                    ans++;
                }
            }
        }
        return ans;
    }
};

```

## 3642 垂直翻转子矩阵

> 垂直翻转 = 沿着水平轴做镜像，实现方式就是交换对称的行

```cpp
vector<vector<int>> reverseSubmatrix(vector<vector<int>>& grid, int x, int y, int k){
    int l = x, r = x + k - 1;
    while(l < r){
        for(int j = y; j < y + k; j++){
            swap(grid[l][j], grid[r][j]);
        }
        l++;
        r--;
    }
    return grid;
}

## 1886 判断矩阵论转后是否一致

> 矩阵顺时针旋转90的处理方式
>> 先以对角线对掉，然后按行翻转

```text
(i, j) --> 顺时针旋转90度  (j, n - 1 - i)

先按照 对对角线对掉
(i, j)  --> (j, i)

再发转每行
(j, i) --> (j, n - 1- i)

```

```cpp
class Solution {
public:
    void rotate(vector<vector<int>>& matrix){
        int n = matrix.size();
        for(int i = 0; i < n; i ++){
            for(int j = i + 1; j < n; j++){
                // 先沿着对角线交换
                swap(matrix[i][j], matrix[j][i]);
            }
            ranges::reverse(matrix[i]);     // 行翻转
        }
    }
    bool findRotation(vector<vector<int>>& mat, vector<vector<int>>& target) {
        // 注意转四次就回到原点了
        for(int i = 0; i < 4; i++){
            if(mat == target){
                return true;
            }else{
                rotate(mat);
            }
        }
        return false;
    }
};

## 1594 矩阵的最大非负积


```cpp
class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        long long MOD = 1e9 + 7;

        // 使用 long long 防止中间计算溢出
        vector<vector<long long>> max_dp(m, vector<long long>(n));
        vector<vector<long long>> min_dp(m, vector<long long>(n));

        // 初始化起点
        max_dp[0][0] = min_dp[0][0] = grid[0][0];

        // 初始化第一行
        for (int j = 1; j < n; ++j) {
            max_dp[0][j] = min_dp[0][j] = max_dp[0][j - 1] * grid[0][j];
        }

        // 初始化第一列
        for (int i = 1; i < m; ++i) {
            max_dp[i][0] = min_dp[i][0] = max_dp[i - 1][0] * grid[i][0];
        }

        // 填充 DP 表
        for (int i = 1; i < m; ++i) {
            for (int j = 1; j < n; ++j) {
                long long v = grid[i][j];
                if (v >= 0) {
                    max_dp[i][j] = max(max_dp[i - 1][j], max_dp[i][j - 1]) * v;
                    min_dp[i][j] = min(min_dp[i - 1][j], min_dp[i][j - 1]) * v;
                } else {
                    max_dp[i][j] = min(min_dp[i - 1][j], min_dp[i][j - 1]) * v;
                    min_dp[i][j] = max(max_dp[i - 1][j], max_dp[i][j - 1]) * v;
                }
            }
        }

        long long res = max_dp[m - 1][n - 1];
        // 如果最终最大积小于 0，按题意返回 -1
        return res < 0 ? -1 : res % MOD;
    }
};

```
