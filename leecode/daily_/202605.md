### 1345 跳跃游戏IV 

这是一个 BFS 类型的算法，讲究的是广度优先，任何一个但钱可选的路线顺着这个路线一直往下走优先到达结果即可结束。


```cpp
class Solution {
public:
    int minJumps(vector<int>& arr) {
        
        int n = arr.size();

        unordered_map<int, vector<int>> valueToIndex;

        for(int i = 0; i< n; i++){
            valueToIndex[arr[i]].push_back(i);
        }

        vector<bool> visited(n, false);

        // 用队列实现广度优先搜索的层级遍历，从而保证第一次到达最后一个位置时所需要的步数时最少的。退出的条件是，队列为空，无法到达最后一个节点，或者已经到达最后一个节点，返回结果为所用最少步数
        queue<int> q;
        q.push(0);

        visited[0] = true;
        int steps = 0;

        while(!q.empty()){
            int size = q.size();
            while(size--){
                int cur = q.front(); q.pop();
                if(cur == n - 1)return steps;

                // 同值跳跃
                int val = arr[cur];
                if(valueToIndex.count(val)){
                    for(auto idx : valueToIndex[val]){
                        if(!visited[idx]){
                            visited[idx] = true;
                            q.push(idx);
                        }
                    }

                    valueToIndex.erase(val);
                }


                // 相邻跳跃
                if(cur - 1 >= 0 && !visited[cur - 1]){
                    visited[cur - 1] = true;
                    q.push(cur - 1);
                }
                if(cur + 1 < n && !visited[cur + 1]){
                    visited[cur + 1] = true;
                    q.push(cur + 1);
                }


                
            }
            ++steps;
        }

        return -1;
    }
};

```
