### 3741 三个相等元素之间的最小距离II

求三个相等元素的两两索引距离，其实就是求 最大的索引位置与最小索引位置距离的两倍

> for(auto& [_, pos] : pos)   这个新特性
> > 这是 c++17 结构绑定新特性

```cpp

int minimumDistance(vector<int>& nums){
  unordered_map<int, vector<int>> pos;
  for(int i = 0; i < nums.size(); i++){
    pos[nums[i]].push_back(i);
  }

  int ans = INT_MAX;
  for(auto& [_, p] : pos){
    for(int i = 2; i < p.size(); i++){
      ans = min(ans, (p[i] - p[i - 2]) * 2）；
    }
  }

  return ans == INT_MAX ? -1 : ans;

}
```


### 1855 下标对中的最大距离


```cpp
class Solution {
public:
    int maxDistance(vector<int>& nums1, vector<int>& nums2) {
        int ans = 0;
        int i = 0;
        for(int j = 0; j < nums2.size(); j++){
          while(i < nums1.size() && nums1[i] > nums2[j]){
            i++;
          }
          if(i == nums1.size()){
            break;
          }
          ans = max(ans, j - i);
        }
        return ans;
    }
};
```

### 1391 检查网格中是否存在有效路径

- bfs
- [并查集](https://oi-wiki.org/ds/dsu/)

```cpp
class Solution {
public:

    vector<int> parent;

    int find(int x){
        return parent[x] == x ? x : parent[x] = find(parent[x]);
    }

    void unite(int x, int y){
        parent[find(x)] = find(y);
    }
    bool hasValidPath(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();

        parent.resize(m * n);

        iota(parent.begin(), parent.end(), 0);

        auto hasRight = [](int s){return s == 1 || s == 4 || s == 6;};
        auto hasLeft  = [](int s){return s == 1 || s == 3 || s == 5;};
        auto hasDown  = [](int s){return s == 2 || s == 3 || s == 4;};
        auto hasUp    = [](int s){return s == 2 || s == 5 || s == 6;};

        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                int cur = i * n + j;
                
                // 和右邻居合并
                if(j + 1 < n && hasRight(grid[i][j]) && hasLeft(grid[i][j + 1])){
                    unite(cur ,cur + 1);
                }

                // 和下邻居合并
                if(i + 1 < m && hasDown(grid[i][j]) &&  hasUp(grid[i + 1][j])){
                    unite(cur, cur + n);
                }
            }
        }

        return find(0) == find(n * m - 1);
    }
};

```


### 2033 获取单值网格的最小操作数

```cpp
class Solution {
public:
    int minOperations(vector<vector<int>>& grid, int x) {
        vector<int> nums;
        for(auto val : grid){
            for(auto v: val){
                nums.push_back(v);
            }
        }

        int mod = nums[0] % x;
        for(auto num : nums){
            if(num % x != mod){
                return -1;
            }
        }

        sort(nums.begin(), nums.end());
        int mid = nums[nums.size() / 2];
        int cos = 0;
        for(auto num : nums){
            cos += abs(num - mid) / x;
        }
        return cos;
    }
};
```
