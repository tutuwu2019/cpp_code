```cpp
class Solution {
public:
    vector<vector<string>> solveNQueens(int n) {
        
        vector<string> board(n, string(n, '.'));
        vector<vector<string>> result;

        vector<bool> col(n, false);
        vector<bool> diag1(2 * n - 1, false);
        vector<bool> diag2(2 * n - 1, false);

        dfs(n, 0, board, col, diag1, diag2, result);

        return result;
    }

    void dfs(int n, int row, vector<string>& board, vector<bool>& col, vector<bool>&diag1, vector<bool>& diag2, vector<vector<string>>& result){

        if(n == row){
            result.push_back(board);
            return;
        }

        for(int i = 0; i < n; i++){
            
            if(col[i] || diag1[row - i + n - 1] || diag2[row + i]){
                continue;
            }

            col[i] = true;
            diag1[row - i + n - 1] = true;
            diag2[row + i] = true;
            board[row][i] = 'Q';
            dfs(n, row + 1, board, col, diag1, diag2, result);

            col[i] = false;
            diag1[row - i + n - 1] = false;
            diag2[row + i] = false;
            board[row][i] = '.';
        }

        return;
    }
};
```

```cpp
class Solution {
public:
    vector<vector<string>> solveNQueens(int n) {
        vector<vector<string>> result;
        vector<int> queens(n, -1); // queens[row] = col
        vector<string> board(n, string(n, '.'));
        dfs(0, n, 0, 0, 0, queens, board, result);
        return result;
    }

private:
    void dfs(int row, int n, int cols, int diag1, int diag2,
             vector<int>& queens, vector<string>& board,
             vector<vector<string>>& result) {
        if (row == n) {
            result.push_back(board);
            return;
        }
        // 得到当前行所有可放置列的位置（1表示空闲，0表示被占）
        int available = ((1 << n) - 1) & ~(cols | diag1 | diag2);
        while (available) {
            int colBit = available & -available;          // 取最低位的1
            int col = __builtin_ctz(colBit);              // 得到列索引
            queens[row] = col;
            board[row][col] = 'Q';

            dfs(row + 1, n,
                cols | colBit,
                (diag1 | colBit) << 1,   // 下一行的主对角线左移一位
                (diag2 | colBit) >> 1,   // 下一行的副对角线右移一位
                queens, board, result);

            board[row][col] = '.';
            available &= available - 1;   // 去掉最低位的1
        }
    }
};
```

### 35 搜索插入位置

- 左闭右开
>  左闭右开 [left, right)
> > 包含左边界 left 指向的元素
> > 不包含有边界 right 指向的元素
> >  区间内的元素下标满足 left <= index < right
> >  区间长度(元素个数) right - left 

语义优势：
1. 长度计算简洁：区间[l, r) 的长度直接是 r - l ,无需 + 1
2. 空去见表示自然： 当 l == r 时，区间为空，无需额外判断
3. 分割时无重叠：将区间[l, r) 分成 [l, mid) 和 [mid, r) 时，两个子区间恰好能覆盖原区间，且不重叠
4. 与循环边界契合：遍历数组时，常用 for(int i = 0; i < n; i++) 这正是 [0, n) 的写法
5. 避免 off-by-one 错误，很多二分法、分治算法采用左闭右开 可以显著减少边界调整的混乱

常用场景：
1. 二分查找
2. 滑动窗口(固定窗口、可变窗口)
3. 归并排序
4. 快速排序(分区)
5. 线段树/树状数组
6. 前缀和与差分
7. 循环不变式设计

** 二分推理的过程充满一种快要获得成功的喜感**

```cpp
class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int left = 0, right = nums.size();

        /*
            使用维持一种左闭右开的状态，
            1. 当 nums[mid] < target 时，mid 以及左侧都不能是答案，所以left = mid + 1
            2. 当 nums[mid] >= target 时，mind 以及左侧都是答案，会选择索引最小的那个答案，所以 right = mid

            循环结束  left >= right  找到那个索引最小的 值
        */
        while(left < right){
            int mid = left + (right - left) / 2;

            if(nums[mid] < target){
                left = mid + 1;
            }else{
                right = mid;
            }
        }
        return left;
    }
};
```

```cpp
public:
    bool searchMatrix(vector<vector<int>>& matrix, int target) {
        if(matrix.empty() || matrix[0].empty()) return false;

        int m = matrix.size(), n = matrix[0].size();

        int left = 0, right = n * m;

        while(left < right){
            int mid = left + (right - left ) / 2;

            int mid_val = matrix[mid / n][mid % n];

                return true;
            }
            else if(mid_val < target){
                left = mid + 1;
            }else{
                right = mid;
            }
        }
        return false;
    }
};
```

```cpp
class Solution {
    int lower_bound(vector<int> &nums,int target){
        int left = 0, right = nums.size();
        
        while(left < right){
            int  mid = left + (right - left) / 2;
            if(nums[mid] < target){
                left = mid +1;
            }
            else
                right = mid;
        }
        return left;
    }

    // 找到第一个 大于 target 的 index 然后 index - 1  可以等价用 lwo_bound(nums, target + 1) -1
    int upper_bound(vector<int>& nums, int target){
        int left = 0, right = nums.size();

        while(left < right){
            int mid = left + (right - left) / 2;
            
            if(nums[mid] <= target){
                left = mid + 1;
            }else{
                right = mid;
            }
        }
        return left - 1;
    }

public:
    vector<int> searchRange(vector<int>& nums, int target) {
        int start = lower_bound(nums,target);
        if(start == nums.size() || nums[start]!= target)
            return {-1,-1};
        int end = upper_bound(nums, target);
        return {start,end};

    }
};
```

#### 示例 1：target 存在于数组中

**输入**：`nums = [1, 3, 5, 7, 9]`，`target = 5`
**期望**：返回索引 `3`（值为 `7`，即第一个大于 5 的位置）

| Step | left | right | mid | nums[mid] | 条件                  | 操作        | 新区间   |
| ---- | ---- | ----- | --- | --------- | --------------------- | ----------- | -------- |
| 1    | 0    | 5     | 2   | 5         | `nums[2]=5 <= target` | `left = 3`  | `[3, 5)` |
| 2    | 3    | 5     | 4   | 9         | `nums[4]=9 > target`  | `right = 4` | `[3, 4)` |
| 3    | 3    | 4     | 3   | 7         | `nums[3]=7 > target`  | `right = 3` | `[3, 3)` |
| 结束 | 3    | 3     | -   | -         | `left == right` 退出  | 返回 `3`    | ✅ 正确  |

> **注意**：在 Step 3 中，`mid = 3`，`nums[3] = 7 > 5`，我们设置 `right = 3`，此时区间变为 `[3, 3)` 空区间，循环结束。返回 `left = 3`，正是答案。这里 `mid = 3` 没有被新区间包含，但最终 `left` 停留在了 `3`。

### 搜索旋转排序数组

```cpp
class Solution {
public:
    int search(vector<int>& nums, int target) {
        int left = 0, right = nums.size();          // 区间 [left, right)
        while (left < right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] == target) return mid;
            
            // 判断左半 [left, mid) 是否有序（因为 nums[left] <= nums[mid-1] 等价于 nums[left] <= nums[mid] ？注意区间开闭）
            // 实际上，左半有序的条件是 nums[left] <= nums[mid]，因为 mid 是当前区间的中点（右开，所以左半是 [left, mid)）
            // 但注意：当区间长度为2时，mid = left，此时 nums[left] <= nums[mid] 恒成立，但左半只有一个元素，肯定有序。
            // 这里沿用闭区间的判断逻辑：若 nums[left] <= nums[mid]，则左半有序；否则右半有序。

            // 注意 这个 等号 的颗粒度
            if (nums[left] <= nums[mid]) {
                // 左半有序
                if (nums[left] <= target && target < nums[mid]) {
                    right = mid;                     // target 在左半 [left, mid)
                } else {
                    left = mid + 1;                  // target 在右半 [mid+1, right)
                }
            } else {
                // 右半有序 (mid, right) 注意：这里右半实际上是 [mid+1, right)，因为 nums[mid] 已经比较过
                // 需要判断 target 是否在右半有序区间内：nums[mid] < target && target <= nums[right-1]
                // 但注意 right 是开边界，所以最后一个元素索引是 right-1。
                if (nums[mid] < target && target <= nums[right- 1]) {
                    left = mid + 1;                  // target 在右半 [mid+1, right)
                } else {
                    right = mid;                     // target 在左半 [left, mid)
                }
            }
        }
        return -1;
    }
};

```


### 153 寻找旋转排序数组中的最小值


```cpp
class Solution {
public:
    int findMin(vector<int>& nums) {
        int n = nums.size();
        int target = nums.back();  // nums.back()  一定归属 较小那段
        int left = 0, right = n;

        // 二分 先准归属块(较小的那个块)，然后在这个块中找到最小的值(这段的起始点)
        while(left < right){
            int mid = left + (right - left) / 2;
            
            if(nums[mid] <= target){
                right = mid;
            }else{
                left = mid + 1;
            }
        }
        return nums[left];
    }
};

```

```cpp

class Solution {
public:
    string decodeString(string s) {
        int num;
        string current;
        stack<int> numStack;
        stack<string> strStack;

        for(auto& ch : s){
            if(isdigit(ch)){
                num = num * 10 + ch - '0';
            }else if(ch == '['){
                numStack.push(num);
                strStack.push(current);
                num = 0;
                current.clear();        // 在遇到 [  的时候 current 会入栈，然后清零
             }else if(ch == ']'){
                int repeated = numStack.top(); numStack.pop();
                string pre = strStack.top(); strStack.pop();

                string str;
                for(int i = 0; i < repeated; i++){
                    str += current;
                }
                current = pre + str;        // 遇到 ] 的时候 current 会重新拼接
            }else{
                current += ch;
            }
        }
        return current;

    }
};
```

```cpp
class Solution {
public:
    vector<int> dailyTemperatures(vector<int>& temperatures) {
        int n = temperatures.size();
        stack<int> sk;

        vector<int> ans(n, 0);

        for(int i = 0; i < n; i++){
            while(!sk.empty() && temperatures[i] > temperatures[sk.top()]){
                int preIndex = sk.top();
                sk.pop();

                ans[preIndex] = i - preIndex;
```

```cpp
class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        // 前后补0，作为哨兵，统一处理边界
        heights.insert(heights.begin(), 0);
        heights.push_back(0);
        stack<int> st; // 存储下标，栈内高度单调递增
        int maxArea = 0;

        for (int i = 0; i < heights.size(); ++i) {
            // 当前高度小于栈顶高度时，栈顶柱子的右边界就是 i，左边界是栈顶下一个元素
            while (!st.empty() && heights[i] < heights[st.top()]) {
                int h = heights[st.top()]; st.pop();
                int left = st.top(); // 左边界（第一个比 h 矮的柱子）
                int width = i - left - 1;
                maxArea = max(maxArea, h * width);
            }
            st.push(i);
        }
        return maxArea;
    }
```
