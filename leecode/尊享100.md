### leetcode 1427 字符串的左右移

关键词：三倍字符串  s + s + s (两倍字符串构建了一个全选换展示窗，从而避免频繁的字符串切片或拼接操作)

左移: 相当于字符串上向右移动起点   
右移：相当于字符串上向左移起点

> 理解为什么 左移 实际上是 起点右移
想象字符串 abc 是一节节车厢，串在一起变成了无限长的传送带：
... abc | abc | abc ...
左移（Left Shift 1 步）
  - 直观动作： 把 abc 变成 bca。
  - 算法视角： 字符串不动，我们的眼睛（窗口）向右挪了一格。
  - 结果： 原来看到的是 [abc]，眼睛向右挪后，看到的是 a[bca]bc。
  - 结论： 字符串向左移 $x$，等于截取窗口的起点向右移动 $x$。

```cpp
class Solution {
public:
    string stringShift(string s, vector<vector<int>>& shift) {
       int n = s.size(), move = 0;
       s = s + s + s;
       for(auto st : shift){
         move -= (st[0] * 2 - 1) * (st[1] % n);
    }
      return s.substr(n + move % n, n);
};

```

### leetcode 161 相隔为1 的距离

```cpp
class Solution {
public:
    bool isOneEditDistance(string s, string t) {
        /*
            数据清理，s.size() <= t.size()，且 如果 s.size() < t.size() 必须 s.size() + 1 = t.size()
            当 s[i] != t[i] 
                如果 s.size() == t.size() 比对 s.substr(i + 1) 是否和 t.substr(i + 1) 一样
                否则， s.substr(i ) 是否和 t.substr(i  +1) 一样
            如果遍历完后还没出结果，则 s 是 t 的前缀 直接 s.size() + 1 == t.size()
        */

        int ns = s.size();
        int nt = t.size();
        if(ns > nt){
            return isOneEditDistance(t, s);
        }
        if(nt - 1 > ns){      // 注意语义的理解： 如果改成 nt - 1 == ns  则会把替换的那部分给遗漏掉  属于剪枝上的一个小差错
            return false;
        }
        for(int i = 0; i < ns; i++){
             if(s[i] != t[i]){
                if(ns == nt){
                    return s.substr(i + 1) == t.substr(i + 1); 
                }else{
                    return s.substr(i) == t.substr(i + 1);
                }
             }
        }
        return ns + 1 == nt;
    }
};

```

```cpp
class Solution {
public:

    /*
        辅助函数，反转指定闭区间 [i, j]
    */
    void reverseRange(vector<char>& s, int i, int j){
        while(i < j){
            swap(s[i++], s[j--]);
        }
    }

    /*
        单词反转一次，整个vector 反转一次，负负得正，
    */
    void reverseWords(vector<char>& s) {
        int n = s.size();
        // 1. 反转每个独立的单词
        int start = 0;
        for(int end = 0; end <= n; ++end){
            if(end == n || s[end] == ' '){
                reverseRange(s, start, end - 1);
                start = end + 1;
            }
        }

        reverseRange(s, 0, n - 1);
    }
};
```

```cpp
class Solution {
public:
    int shortestWay(string source, string target) {
        int res = 0;
        int t_idx = 0;
        while(t_idx < target.size()){
            // 记录这一轮开始前的进度
            int pre_t_idx = t_idx;

            for(int s_idx = 0; s_idx < source.size(); s_idx++){
                if(t_idx < target.size() && source[s_idx] == target[t_idx]){
                    t_idx++;
                }
            }
            if(pre_t_idx == t_idx){
                // 说明 t_idx 没有动，也就是没匹配到
                return -1;
            }
            res++;
        }
        return res;
    }
};
```

### leetcode 159 至多包含两个字符的最长字符串

> 注意：目标字串最多只能有两个字符串。
> > 采用滑动窗口加 哈希表维护窗口中的元素。当元素个数超过2，元素的个数逐步减一，如果减到最后元素为空，删除这个元素的记录。
> 完成一轮新的 统计 右边界右移

```cpp
class Solution {
public:
    int lengthOfLongestSubstringTwoDistinct(string s) {
        int n = s.size();
        if(n < 3){
            return n;
        }

        int left = 0, right = 0, max_len = 2;
        unordered_map<char, int> hash_map;
        while(right < n){
            hash_map[s[right]]++;

            while(hash_map.size() > 2){
                hash_map[s[left]]--;
                if(hash_map[s[left]] == 0){
                    hash_map.erase(s[left]);
                }
                left++;
            }
            max_len = max(max_len, right - left + 1);
            right++;
        }
        return max_len;
    }
};
```

```cpp
class Solution {

        int n = s.size();
```

```cpp
        int left = 0, right = 0, count = 0;

                repeat_count++;

                }
                left++;

                count++;
            }
            right++;
        }
        return count;
```

```cpp
bool canPermutePalindrome(string s) {
        /*
            1. 字符串个数为偶数必然
            2. 字符串个数为奇数，统计所有偶数的字符和是否为偶数  ==> 奇数字符不能超过1
        */
        /*
        int counts[128] = {0};

        for(char c : s){
            counts[c]++;
        }

        int odd = 0;
        for(int i = 0; i < 128; i++){
            if(counts[i] % 2 != 0) odd++;
        }
        return odd <= 1;
        */
        bitset<128> bits;
        for(char c : s){

        }
```

### leetcode 734 句子相似性

> 句子相似性：就是两个句子的单词个数要相同，相同的 idx 单词相同好继续比对下一个单词，如果比对的不同，就查相似表中有没有相关记录(相似规则)如果有那也进行下一个

```cppp
bool areSentencesSimilar(vector<string>& sentence1, vector<string>& sentence2, vector<vector<string>>& similarPairs) {
        /*
            两个句子相似，必须同时满足两个条件：1.长度相同，2.满足"单词"相似规则(即相似的映射规则，比如例1中 great 与 find 相似)
        
            注意：比对的是相对的位置(idx 相对一致)
        */

        if(sentence1.size() != sentence2.size()){
            return false;
        }

        unordered_set<string> s;
        for(const auto& pair : similarPairs){
            s.insert(pair[0] + "#" + pair[1]);
        }

        for(int i = 0; i < sentence1.size(); i++){
            string w1 = sentence1[i];
            string w2 = sentence2[i];

            if(w1 != w2 && !s.count(w1 + "#" + w2) && !s.count(w2 + "#" + w1)){
                return false;
            }
        }
        return true;
    }

```

### leetcode 移动字符串分组

> 建模 如何归一化再分组? ==> 为每一组找一个 代表元
> abc、bcd、cde 的特点都是相邻字符之间的相对距离是恒定的  所以可以归一化 统一用 abc 来表示，为了同一，所有的代表元都的首个字符都从 'a' 开始
> > 如果是循环移位怎么处理？ 取模  比如 az (z-a=25)、ba(b-a=-1  (-1 + 26) % 26 = 25)

```cpp
vector<vector<string>> groupStrings(vector<string>& strings) {
        unordered_map<string, vector<string>> groups;

        for(auto& str : strings){
            int offset = str[0] - 'a';
            string key = "";

            for(auto ch : str){
                char nomarlized_char = (ch - 'a' + 26 - offset) % 26 + 'a';     // - offset 是位于让 统一元 的首字符相对a
                key += nomarlized_char;
            }
            groups[key].push_back(str);

        }

        vector<vector<string>> res;
        for(auto& str : groups){
            res.push_back(str.second);
        }
        return res;
    }
```

```cpp
vector<vector<int>> multiply(vector<vector<int>>& mat1, vector<vector<int>>& mat2) {
        int m = mat1.size();
        int k = mat1[0].size();
        int n = mat2[0].size();

        vector<vector<int>> res(m, vector<int>(n, 0));

        for(int i = 0; i < m; i++){
            for(int p = 0; p < k; p++){
                if(mat1[i][p] != 0){
                    for(int j = 0; j < n; j++){
                        if(mat2[p][j] != 0){
                            res[i][j] += mat1[i][p] * mat2[p][j];
                        }
                    }
                }
            }
        }
        return res;
```

```cpp
vector<vector<int>> candyCrush(vector<vector<int>>& board) {
        int m = board.size();
        int n = board[0].size();
        bool todo = false;
        // 先标记水平
        for(int r = 0; r < m; r++){
            for(int c = 0; c + 2 < n; c++){
                int val = abs(board[r][c]);
                if(val != 0 && val == abs(board[r][c + 1]) && val == abs(board[r][c + 2])){
                    board[r][c] = board[r][c + 1] = board[r][c + 2] = -val;
                    todo = true;
                }
            }
        }

        // 标记垂直
        for(int r = 0; r + 2 < m; r++){
            for(int c = 0; c < n; c++){
                int val = abs(board[r][c]);
                if(val != 0 && val == abs(board[r + 1][c]) && val == abs(board[r + 2][c])){
                    board[r][c] = board[r + 1][c] = board[r + 2][c] = -val;
                    todo = true;
                }
            }
        }

        //粉碎与下落
        if(todo){
            for(int c = 0; c < n; ++c){
                int wr = m - 1;
                for(int r = m - 1; r >= 0; r--){
                    if(board[r][c] > 0){
                        board[wr--][c] = board[r][c];  // 不断的把元素下移
                    }
                }
                while(wr >= 0){
                    board[wr--][c] = 0; //已经下移后，原来的位置全部用0补充
                }
            }
            // 递归处理，直到棋盘稳定
            return candyCrush(board);
        }
        return board;
    }
```

```cpp
vector<vector<int>> candyCrush(vector<vector<int>>& board) {

        /*
            我们将递归改为一个 while(true) 循环，利用一个布尔变量 changed 来标记当前回合是否发生了 “粉碎”。如果某一回合没有任何糖果被标记，说明棋盘已稳定，退出循环。
        */
        int m = board.size();
        int n = board[0].size();

        while (true) {
            bool changed = false;
            
            // 1. 水平扫描标记
            for (int r = 0; r < m; ++r) {
                for (int c = 0; c + 2 < n; ++c) {
                    int v = abs(board[r][c]);
                    if (v != 0 && v == abs(board[r][c + 1]) && v == abs(board[r][c + 2])) {
                        board[r][c] = board[r][c + 1] = board[r][c + 2] = -v;
                        changed = true;
                    }
                }
            }

            // 2. 垂直扫描标记
            for (int r = 0; r + 2 < m; ++r) {
                for (int c = 0; c < n; ++c) {
                    int v = abs(board[r][c]);
                    if (v != 0 && v == abs(board[r + 1][c]) && v == abs(board[r + 2][c])) {
                        board[r][c] = board[r + 1][c] = board[r + 2][c] = -v;
                        changed = true;
                    }
                }
            }

            // 3. 如果本轮没有发生粉碎，退出循环
            if (!changed) break;

            // 4. 执行重力下落 (Gravity Drop)
            // 这一步直接覆盖负数标记，实现“粉碎”与“下落”合一
            for (int c = 0; c < n; ++c) {
                int wr = m - 1; // 写指针：从底部开始往上填
                for (int r = m - 1; r >= 0; --r) {
                    if (board[r][c] > 0) { // 只有正数（未被粉碎）的糖果保留
                        board[wr--][c] = board[r][c];
                    }
                }
                // 将上方剩余空间全部置零
                while (wr >= 0) {
                    board[wr--][c] = 0;
```

```cpp
int minMeetingRooms(vector<vector<int>>& intervals) {
        // 用最小堆存储会议室的结束时间

        if(intervals.empty()){
            return 0;
        }

        sort(intervals.begin(), intervals.end(), [](const vector<int>& a, const vector<int>& b){return a[0] < b[0];});

        // 小根堆的对顶是最找结束会议的时间
        priority_queue<int, vector<int>, greater<int>> pq;

        pq.push(intervals[0][1]);

            if(intervals[i][0] >= pq.top()){
                pq.pop();
            }
            // 每一个会议的结束时间都要记录
            pq.push(intervals[i][1]);
        }

        return pq.size();
    }
```

```cpp
int mostBooked(int n, vector<vector<int>>& meetings) {
    // 1. 初始准备
    vector<int> count(n, 0); // 记录每个房间的使用次数
    sort(meetings.begin(), meetings.end()); // 按开始时间排序

    // 空闲堆：存房间号 (0, 1, ..., n-1)
    priority_queue<int, vector<int>, greater<int>> freeRooms;
    for (int i = 0; i < n; ++i) freeRooms.push(i);

    // 忙碌堆：存 {结束时间, 房间号}
    typedef pair<long long, int> PLI;
    priority_queue<PLI, vector<PLI>, greater<PLI>> busyRooms;

    for (auto& m : meetings) {
        long long start = m[0], end = m[1];
        
        // A. 释放所有在当前会议开始前已经结束的房间
        while (!busyRooms.empty() && busyRooms.top().first <= start) {
            freeRooms.push(busyRooms.top().second);
            busyRooms.pop();
        }

        if (!freeRooms.empty()) {
            // B. 有空闲房间：直接分配编号最小的
            int room = freeRooms.top();
            freeRooms.pop();
            count[room]++;
            busyRooms.push({end, room});
        } else {
            // C. 没空闲房间：等最早结束的房间
            auto [earliestEnd, room] = busyRooms.top();
            busyRooms.pop();
            count[room]++;
            // 新的结束时间 = 房间腾出来的时间 + 会议持续时间
            long long newEnd = earliestEnd + (end - start);
            busyRooms.push({newEnd, room});
        }
    }

    // 3. 找出使用次数最多的房间
    int maxIdx = 0;
    for (int i = 1; i < n; i++) {
        if (count[i] > count[maxIdx]) maxIdx = i;
    }
    return maxIdx;
}
```

## 708 链表循环有序插入

> 三种情况：空、正常插入、反转处

```cpp
/*
// Definition for a Node.
class Node {
public:
    int val;
    Node* next;

    Node() {}

    Node(int _val) {
        val = _val;
        next = NULL;
    }

    Node(int _val, Node* _next) {
        val = _val;
        next = _next;
    }
};
*/

class Solution {
public:
    Node* insert(Node* head, int insertVal) {
        Node* node = new Node(insertVal);

        if(head == nullptr){
            //head->next = node;  空节点不能这样用
            node->next = node;
            return node;
        }

        Node* pre = head;
        Node* cur = head->next;

        // 注意边界条件不是 cur->next 而是 cur != head
        while(cur != head){
            if(pre->val <= insertVal && insertVal <= cur->val){
                break;
            }

            if(pre->val > cur->val){
                // 这里的过滤逻辑是：insertVal >= pre->val || insertVal <= cur->val
                if(pre->val <= insertVal || insertVal <= cur->val){
                    break;
                }
            }
            pre = cur;
            cur= cur->next;
            
        }
        pre->next = node;
        node->next = cur;
        return head;
    }
};
```

## 369 给单链表加1

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    // 最右的非9 找到以后（noteNode），后面的所有节点的val 都要加1 是否进位 是由个noteNode 决定的
    ListNode* plusOne(ListNode* head) {
        ListNode* dummy = new ListNode(0, head);

        ListNode* cur = head;
        ListNode* noteNode = dummy;

        while(cur){
            if(cur->val != 9){
                noteNode = cur;
            }
            cur = cur->next;
        }

        noteNode->val++;
        cur = noteNode->next;
        while(cur){
            cur->val = 0;
            cur = cur->next;
        }

        return dummy->val == 1 ? dummy : head;
    }
};
```
