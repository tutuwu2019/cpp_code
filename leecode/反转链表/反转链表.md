### leetcode 209 反转链表

> 迭代

采用头插法
> 注意实际开发记得释放内存

```cpp
ListNode* reverseList(ListNode* head) {
        ListNode* pre = nullptr;
        ListNode* curr = head;
        ListNode* tmp = nullptr;
        while(curr){
            tmp = curr->next; // 不建议在循环里面 new auto tmp = curr->next;
            curr->next = pre;
            pre = curr;
            curr = tmp;
        }
        return pre;
    }
```

### 92 反转链表II

> 反转 区间的节点，依然采用头插法

```cpp
ListNode* reverseBetween(ListNode* head, int left, int right) {
        ListNode* dummy = new ListNode(0, head);
        ListNode* pre = dummy;
        for(int i = 0; i < left - 1; i++){
            pre = pre->next;
        }
        ListNode* cur = pre->next;
        ListNode* tmp = nullptr;

        // 因为 这是区间反转， 所以建模核心 不变的其实是 tmp  而不像 整个链表反转 不变的是 cur
        // 需要隔离 tmp->next 、维护 pre
        for(int i = 0; i < right - left; i++){
            tmp = cur->next;            // 暂存 next 节点
            cur->next = tmp->next;      // 确定 cur 后面的链表
            tmp->next = pre->next;      // next->next 接 pre->next
            pre->next = tmp;            // 把 next 插到 pre->next
        }

        return dummy->next;
    }

```

### leetcode 26 k个一组反转链表

> 逻辑叙述完整，每个变量的作用、定位

```cpp
pair<ListNode*, ListNode*> reverse(ListNode* head, ListNode* tail){
        ListNode* pre = nullptr;
        ListNode* cur = head;
        ListNode* tmp = nullptr;

        while(cur != tail){
            tmp = cur->next;
            cur->next = pre;
            pre = cur;
            cur = tmp;
        }
        return {pre, head};
    }
    ListNode* reverseKGroup(ListNode* head, int k) {
        if(head == nullptr || k == 1){
            return head;
        }

        ListNode* dummy = new ListNode(0, head);

        ListNode* pre = dummy;

        while(pre->next){
            ListNode* start = pre->next;
            ListNode* end = pre;

            for(int i = 0; i < k; i++){
                end = end->next;
                if(end == nullptr){
                    // 后面的不用管
                    return dummy->next;
                }
            }

            ListNode* nextGroupHead = end->next;

            pair<ListNode*, ListNode*> reverPair = reverse(start, nextGroupHead);

            ListNode* newHead = reverPair.first;
            ListNode* newTail = reverPair.second;

            // pre 拼接 []
            // 更新 pre
            pre->next = newHead;
            newTail->next = nextGroupHead;

            pre = newTail; 
        }

        return dummy->next;
    }

```


### leetcode 24 两两交换链表中的接待你

> 操作有两部，切断 next 与 next->next 的关系，但是记录的 next->next  
>  接着 更新 head->next  
>  把 head 插到 next 后面   “头插法”

```cpp
ListNode* swapPairs(ListNode* head) {
        if(head == nullptr || head->next == nullptr){
            return head;
        }
        ListNode* tmp = nullptr;
        tmp = head->next;

        // head  next 更新
        head->next = swapPairs(tmp->next);

        // 头插法
        // 然后 把 head 插到 tmo 后面，返回 tmp
        tmp->next = head;
        return tmp;
    }

```

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    // 辅助函数反转链表
    ListNode* reverseList(ListNode* head){
        ListNode* cur = head;
        ListNode* pre = nullptr;
        ListNode* tmp = nullptr;

        while(cur){
            tmp = cur->next;
            cur->next = pre;
            pre = cur;
            cur = tmp;
        }
        return pre;
    }
    bool isPalindrome(ListNode* head) {
        if(head == nullptr || head->next == nullptr){
            return true;
        }

        // 采用快慢指针
        ListNode* slow = head;
        ListNode* fast = head;
        while (fast->next != nullptr && fast->next->next != nullptr) {
            slow = slow->next;
            fast = fast->next->next;
        }

        ListNode* secondHalfStart = slow->next;
        slow->next = nullptr;
        ListNode* secondHalfHead = reverseList(secondHalfStart);

        // 比较两者
        ListNode* p1 = head;
        ListNode* p2 = secondHalfHead;

        bool result = true;

        while(p1 != nullptr && p2 != nullptr){
            if(p1->val != p2->val){
                //return false;
                result = false;
                break;
            }
            p1 = p1->next;
            p2 = p2->next;
        }

        // 是否要恢复呢？
        // slow->next = reverseList(secondHalfHead->next);

        return result;
        
    }
};
