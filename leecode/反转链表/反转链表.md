### leetcode 209 反转链表

> 迭代

采用头插法
> 注意实际开发记得释放内存

```cpp
ListNode* reverseList(ListNode* head) {
        if(!head){
            return head;
        }
        ListNode* dummy = new ListNode(0, head);
        
        ListNode* pre = dummy;
        ListNode* cur = head;

        ListNode* tmp = nullptr;
        while(cur->next){
            tmp = cur->next;
            cur->next = tmp->next;
            tmp->next = pre->next;
            pre->next = tmp;
        }
        return dummy->next;     //  或者返回 pre->next 也是一样的  return pre->next;
    }
```

### 92 反转链表II

> 反转 区间的节点，依然采用头插法

```cpp
ListNode* reverseBetween(ListNode* head, int left, int right) {
        ListNode* dummy = new ListNode(0, head);
        ListNode* pre = dummy;
        for(int i = 0; i < left - 1; i++){
            pre = pre->next;
        }
        ListNode* cur = pre->next;
        ListNode* tmp = nullptr;

        // 因为 这是区间反转， 所以建模核心 不变的其实是 tmp  而不像 整个链表反转 不变的是 cur
        // 需要隔离 tmp->next 、维护 pre
        for(int i = 0; i < right - left; i++){
            tmp = cur->next;            // 暂存 next 节点
            cur->next = tmp->next;      // 确定 cur 后面的链表
            tmp->next = pre->next;      // next->next 接 pre->next
            pre->next = tmp;            // 把 next 插到 pre->next
        }

        return dummy->next;
    }

```

### leetcode 25 k个一组反转链表

> 逻辑叙述完整，每个变量的作用、定位

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    // 返回值：反转后的最后一个节点（即下一组的 pre）
ListNode* reverse(ListNode* pre, int k) {
    ListNode* cur = pre->next; 
    for (int i = 0; i < k - 1; i++) {
        ListNode* tmp = cur->next;
        cur->next = tmp->next;
        tmp->next = pre->next;
        pre->next = tmp;
    }
    return cur; // 返回当前的末尾，方便主函数更新 pre
}

ListNode* reverseKGroup(ListNode* head, int k) {
    if (!head || k == 1) return head;
    ListNode* dummy = new ListNode(0, head);
    ListNode* pre = dummy;

    // 先校验长度是否足够 k
    while (true) {
        ListNode* check = pre->next;
        for (int i = 0; i < k; i++) {
            if (!check) return dummy->next;
            check = check->next;
        }
        // 长度足够，执行头插反转
        pre = reverse(pre, k);
    }
}
};
```


### leetcode 24 两两交换链表中的接待你

> 操作有两部，切断 next 与 next->next 的关系，但是记录的 next->next  
>  接着 更新 head->next  
>  把 head 插到 next 后面   “头插法”

```cpp
ListNode* swapPairs(ListNode* head) {
        if(head == nullptr || head->next == nullptr){
            return head;
        }
        ListNode* tmp = nullptr;
        tmp = head->next;

        // head  next 更新
        head->next = swapPairs(tmp->next);

        // 头插法
        // 然后 把 head 插到 tmo 后面，返回 tmp
        tmp->next = head;
        return tmp;
    }

```


### leetcode 234 回文链表

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    bool isPalindrome(ListNode* head) {
        if (!head || !head->next) return true;

        // 1. 快慢指针找中点
        // 结束后，slow 指向前半段的末尾（偶数长度）或中间（奇数长度）
        ListNode* slow = head;
        ListNode* fast = head;
        while (fast->next && fast->next->next) {
            slow = slow->next;
            fast = fast->next->next;
        }

        // 2. 局部头插法翻转后半部分
        // pre 保持在 slow 不动，cur 是后半段的第一个节点
        ListNode* pre = slow;
        ListNode* cur = pre->next;
        while (cur->next) {
            ListNode* tmp = cur->next;
            cur->next = tmp->next;
            tmp->next = pre->next;
            pre->next = tmp;
        }

        // 3. 比较前半段和后半段
        ListNode* p1 = head;
        ListNode* p2 = pre->next; // 翻转后的后半段起点
        while (p2) {
            if (p1->val != p2->val) return false;
            p1 = p1->next;
            p2 = p2->next;
        }

        return true;
    }
};
```


### leetcode 143 重排链表

反转还是和上面的一样，新增了一个合并辅助函数

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    // 辅助函数：反转链表 (迭代法)
    ListNode* reverseList(ListNode* head) {
        ListNode* prev = nullptr;
        ListNode* curr = head;
        while (curr != nullptr) {
            ListNode* nextTemp = curr->next;
            curr->next = prev;
            prev = curr;
            curr = nextTemp;
        }
        return prev;
    }

    // 辅助函数：合并两个链表
    void mergeLists(ListNode* l1, ListNode* l2) {
        while (l1 != nullptr && l2 != nullptr) {
            ListNode* l1_next = l1->next;
            ListNode* l2_next = l2->next;

            l1->next = l2;
            if (l1_next == nullptr) { // 如果 l1 已经到头，l2 直接接上
                break;
            }
            l2->next = l1_next;

            l1 = l1_next;
            l2 = l2_next;
        }
    }

    void reorderList(ListNode* head) {
        if (head == nullptr || head->next == nullptr || head->next->next == nullptr) {
            return;
        }

        // 1. 使用快慢指针找到链表中点
        ListNode* slow = head;
        ListNode* fast = head;
        while (fast->next != nullptr && fast->next->next != nullptr) {
            slow = slow->next;
            fast = fast->next->next;
        }

        // 此时 slow 指向前半部分的最后一个节点
        ListNode* head1 = head;
        ListNode* head2 = slow->next;
        slow->next = nullptr; // 断开前后两部分

        // 2. 反转后半部分链表
        head2 = reverseList(head2);

        // 3. 合并两个链表
        mergeLists(head1, head2);
    }
};

```
