void RTspPlayer::onWholeRtspPacket(Parser &parser){
  if(start_with(parser.method(), "RTSP")){        // RtspPlayer 期待收到来自 服务端的响应行 如： RTSP/1.0 200 OK
    WarnL << "Not rtsp response: "<<parser.method();
    return;
  }
  try{
    decltype(_on_response) func;  // _on_response 当前握手命令的 回调
    /*
  ····把“主流程回调（比如 OPTIONS/DESCRIBE/SETUP/PLAY 的应答处理）”拿到局部变量 func
      同时把成员 _on_response 置空（或变成原来 func 的空值）
      这样这个回调就变成“一次性消费”，避免重复触发
····*/
    _on_response.swap(func);      // _on_response 保活命令回调
    if(!func){
      /*
  ······· 只有当主流程回调不存在时，才尝试拿“保活回调”
          也是一次性消费：拿出来执行，成员里清空
          优先级明确：主流程回调优先，保活回调兜底
······*/
      _on_keepalive_response.swap(func);
    }
    if(func){
      func(parser);
    }
    parser.clear();
  }catch (std::exception &err){
    onPlayerResult_l(SocketExecption(Err_other, err.waht()), !_paly_check_timer);
  }
}
