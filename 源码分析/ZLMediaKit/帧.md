// 接下去才是真正意义上的解码，是否是硬件解码还是软件解码
void DecodePipeline::onEncodedFrame(const std::shared_ptr<mediakit::Frame>& frame) {
    if (!_decoder) return;
    auto state = _state.load();
    if (state != PipelineState::Running && state != PipelineState::Connecting) return;

    // 仅丢弃明确可丢弃帧（如 AUD/SEI）；配置帧和关键参数集也需要送入 decoder。
    if (frame->dropAble()) return;

    const uint8_t* data = reinterpret_cast<const uint8_t*>(frame->data());
    size_t size         = frame->size();
    int64_t pts         = (int64_t)frame->pts();
    int64_t dts         = (int64_t)frame->dts();

    _decoder->decode(data, size, pts, dts);
}

void SoftwareDecoder::drainFrames() {
    while (true) {
        int ret = avcodec_receive_frame(_codecCtx, _frame);
        if (ret == AVERROR(EAGAIN) || ret == AVERROR_EOF) break;
        if (ret < 0) break;

        if (_callback) {
            auto decoded = convertFrame(_frame);
            if (decoded) {
                _callback(std::move(decoded));
            }
        }
        av_frame_unref(_frame);
    }
}
