### LRU cache 
设计和实现一个LRU 缓存机制，支持 get 和 put 操作，要求时间复杂度为o(1)

```cpp
#include <iostream>
#include <unordered_map>
#include <list>

class LRUCache{
private:
  int capacity;
  std::list<std::pair<int, int>> cache;
  std::unordered_map<int, std::list<std::pair<int, int>>::interator> map;

public:
  LRUCache(int cap) : capacity(cap){};

  int get(int key){
    auto it = map.find(key);
    if(it == map::end()){
      return -1;
    }

    cache.splice(cache.begin(), cache, it->second);
    return it->second->second;
  }

  void put(int key, int value){
    auto it = map.find(key);

    if(it != map.end()){
      it->second->second = value;
      cache.splice(cache.begin(), cache, it->second);
      return;
    }

    if(cache.size() == capacity){
      int lastKey = cache.back().first;
      map.erase(lastKey);
      cache.pop_back();
    }

    cache.push_front({ley, value});
    map[key] = cache.begin();
  }

};

int main() {
    LRUCache lru(2); // 容量为 2
    lru.put(1, 1);
    lru.put(2, 2);
    std::cout << lru.get(1) << std::endl;    // 返回 1
    lru.put(3, 3);                         // 导致 key 2 被逐出
    std::cout << lru.get(2) << std::endl;    // 返回 -1 (未找到)
    lru.put(4, 4);                         // 导致 key 1 被逐出
    std::cout << lru.get(1) << std::endl;    // 返回 -1
    std::cout << lru.get(3) << std::endl;    // 返回 3
    std::cout << lru.get(4) << std::endl;    // 返回 4
    return 0;
}

```


### 合并 k 个升序链表

给你一个链表数组，每个链表都已经按照升序排列，请你将所有链表合并到一个升序链表中返回。优先使用队列 (堆) 的使用或者分治的思想，时间复杂度 o (nlogk)

```cpp
#include <iostream>
#include <vector>
#include <queue>

// 链表节点定义
struct ListNode {
    int val;
    ListNode *next;
    ListNode() : val(0), next(nullptr) {}
    ListNode(int x) : val(x), next(nullptr) {}
    ListNode(int x, ListNode *next) : val(x), next(next) {}
};

class Solution {
public:
    // 自定义比较函数：val 越小的优先级越高（放在堆顶）
    struct compare {
        bool operator()(ListNode* a, ListNode* b) {
            return a->val > b->val; 
        }
    };

    ListNode* mergeKLists(std::vector<ListNode*>& lists) {
        // 定义最小堆
        std::priority_queue<ListNode*, std::vector<ListNode*>, compare> pq;

        // 1. 将所有非空链表的头节点放入堆中
        for (auto head : lists) {
            if (head) pq.push(head);
        }

        // 2. 创建虚拟头节点，简化链表拼接
        ListNode dummy(0);
        ListNode* tail = &dummy;

        // 3. 循环取出堆顶元素（当前最小值）
        while (!pq.empty()) {
            ListNode* node = pq.top();
            pq.pop();

            tail->next = node;  // 接到结果链表后
            tail = tail->next;

            // 如果该节点还有下一个节点，将其压入堆中
            if (node->next) {
                pq.push(node->next);
            }
        }

        return dummy.next;
    }
};

// ==================== 测试辅助函数 ====================

// 根据 vector 创建链表
ListNode* createList(const std::vector<int>& nums) {
    if (nums.empty()) return nullptr;
    ListNode* head = new ListNode(nums[0]);
    ListNode* cur = head;
    for (size_t i = 1; i < nums.size(); i++) {
        cur->next = new ListNode(nums[i]);
        cur = cur->next;
    }
    return head;
}

// 打印链表
void printList(ListNode* head) {
    if (!head) {
        std::cout << "NULL" << std::endl;
        return;
    }
    while (head) {
        std::cout << head->val;
        if (head->next) std::cout << " -> ";
        head = head->next;
    }
    std::cout << std::endl;
}

// 释放链表内存
void deleteList(ListNode* head) {
    while (head) {
        ListNode* temp = head;
        head = head->next;
        delete temp;
    }
}

// ==================== Main 函数 ====================

int main() {
    Solution sol;
    
    std::cout << "========== 测试 1: 正常情况 ==========" << std::endl;
    {
        // lists = [[1,4,5],[1,3,4],[2,6]]
        std::vector<ListNode*> lists;
        lists.push_back(createList({1, 4, 5}));
        lists.push_back(createList({1, 3, 4}));
        lists.push_back(createList({2, 6}));
        
        std::cout << "输入链表:" << std::endl;
        for (size_t i = 0; i < lists.size(); i++) {
            std::cout << "  List " << i << ": ";
            printList(lists[i]);
        }
        
        ListNode* result = sol.mergeKLists(lists);
        std::cout << "合并结果: ";
        printList(result);  // 期望: 1 -> 1 -> 2 -> 3 -> 4 -> 4 -> 5 -> 6
        
        deleteList(result);
    }
    
    std::cout << "\n========== 测试 2: 空列表 ==========" << std::endl;
    {
        std::vector<ListNode*> lists;
        ListNode* result = sol.mergeKLists(lists);
        std::cout << "输入: []" << std::endl;
        std::cout << "合并结果: ";
        printList(result);  // 期望: NULL
    }
    
    std::cout << "\n========== 测试 3: 包含空链表 ==========" << std::endl;
    {
        std::vector<ListNode*> lists;
        lists.push_back(nullptr);
        lists.push_back(createList({1, 2}));
        lists.push_back(nullptr);
        lists.push_back(createList({3}));
        
        ListNode* result = sol.mergeKLists(lists);
        std::cout << "合并结果: ";
        printList(result);  // 期望: 1 -> 2 -> 3
        
        deleteList(result);
    }
    
    std::cout << "\n========== 测试 4: 只有一个链表 ==========" << std::endl;
    {
        std::vector<ListNode*> lists;
        lists.push_back(createList({1, 2, 3}));
        
        ListNode* result = sol.mergeKLists(lists);
        std::cout << "合并结果: ";
        printList(result);  // 期望: 1 -> 2 -> 3
        
        deleteList(result);
    }
    
    std::cout << "\n========== 测试 5: 重复元素 ==========" << std::endl;
    {
        std::vector<ListNode*> lists;
        lists.push_back(createList({1, 1, 1}));
        lists.push_back(createList({1, 2}));
        
        ListNode* result = sol.mergeKLists(lists);
        std::cout << "合并结果: ";
        printList(result);  // 期望: 1 -> 1 -> 1 -> 1 -> 2
        
        deleteList(result);
    }
    
    std::cout << "\n所有测试完成!" << std::endl;
    
    return 0;
}
```

分治思想
```cpp
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
        if (!l1) return l2;
        if (!l2) return l1;
        if (l1->val < l2->val) {
            l1->next = mergeTwoLists(l1->next, l2);
            return l1;
        } else {
            l2->next = mergeTwoLists(l1, l2->next);
            return l2;
        }
    }

    ListNode* solve(std::vector<ListNode*>& lists, int left, int right) {
        if (left == right) return lists[left];
        if (left > right) return nullptr;
        int mid = left + (right - left) / 2;
        return mergeTwoLists(solve(lists, left, mid), solve(lists, mid + 1, right));
    }

    ListNode* mergeKLists(std::vector<ListNode*>& lists) {
        return solve(lists, 0, lists.size() - 1);
    }
};
```

### 链表中 k 个元素翻转

```cpp
#include <iostream>

struct ListNode {
    int val;
    ListNode *next;
    ListNode(int x) : val(x), next(nullptr) {}
};

class Solution {
public:
    /**
     * 翻转区间 [a, b) 内的节点（左闭右开）
     */
    ListNode* reverse(ListNode* a, ListNode* b) {
        ListNode* prev = nullptr;
        ListNode* curr = a;
        ListNode* nxt = a;
        while (curr != b) {
            nxt = curr->next;
            curr->next = prev;
            prev = curr;
            curr = nxt;
        }
        // 返回翻转后的新头节点
        return prev;
    }

    ListNode* reverseKGroup(ListNode* head, int k) {
        if (head == nullptr) return nullptr;
        
        // 1. 寻找待翻转区间的终点 b (第 k+1 个节点)
        ListNode* a = head;
        ListNode* b = head;
        for (int i = 0; i < k; i++) {
            // 不足 k 个节点，无需翻转，直接返回
            if (b == nullptr) return head;
            b = b->next;
        }

        // 2. 翻转前 k 个节点
        ListNode* newHead = reverse(a, b);

        // 3. 递归翻转后续链表，并连接起来
        // 此时 a 已经变成了翻转后这组的尾节点
        a->next = reverseKGroup(b, k);

        return newHead;
    }
};

// 辅助函数：打印链表
void printList(ListNode* head) {
    while (head) {
        std::cout << head->val << (head->next ? " -> " : "");
        head = head->next;
    }
    std::cout << std::endl;
}

```

### 排序链表

给定链表的头节点，请将其按照升序排列并返回排序后的链表，要求时间复杂度 o (nlogn)，空间复杂度 o (1)，并写一个 main 测试

```cpp
#include <iostream>
#include <vector>

struct ListNode {
    int val;
    ListNode *next;
    ListNode(int x) : val(x), next(nullptr) {}
};

class Solution {
public:
    // 合并两个升序链表
    ListNode* merge(ListNode* l1, ListNode* l2) {
        ListNode dummy(0);
        ListNode* curr = &dummy;
        while (l1 && l2) {
            if (l1->val < l2->val) {
                curr->next = l1;
                l1 = l1->next;
            } else {
                curr->next = l2;
                l2 = l2->next;
            }
            curr = curr->next;
        }
        curr->next = l1 ? l1 : l2;
        return dummy.next;
    }

    // 自底向上归并排序
    ListNode* sortList(ListNode* head) {
        if (!head || !head->next) return head;

        // 获取链表长度
        int length = 0;
        ListNode* node = head;
        while (node) {
            length++;
            node = node->next;
        }

        ListNode dummy(0);
        dummy.next = head;

        // 自底向上：从 subLen=1 开始，每次翻倍
        for (int subLen = 1; subLen < length; subLen <<= 1) {
            ListNode* prev = &dummy;
            ListNode* curr = dummy.next;

            while (curr) {
                // 找到第一部分 [head1, ...]，长度为 subLen
                ListNode* head1 = curr;
                for (int i = 1; i < subLen && curr->next; ++i) {
                    curr = curr->next;
                }
                
                // 切断第一部分，保存第二部分起点
                ListNode* head2 = curr->next;
                curr->next = nullptr;
                curr = head2;

                // 找到第二部分，长度最多为 subLen
                for (int i = 1; i < subLen && curr && curr->next; ++i) {
                    curr = curr->next;
                }

                // 保存下一组起点，切断第二部分
                ListNode* nextGroup = nullptr;
                if (curr) {
                    nextGroup = curr->next;
                    curr->next = nullptr;
                }

                // 合并当前两部分，并接到 prev 后面
                prev->next = merge(head1, head2);
                
                // 移动 prev 到当前合并后链表的末尾
                while (prev->next) {
                    prev = prev->next;
                }
                
                // 处理下一组
                curr = nextGroup;
            }
        }
        return dummy.next;
    }
};

// ==================== 测试辅助函数 ====================

// 从数组创建链表
ListNode* createList(const std::vector<int>& vals) {
    if (vals.empty()) return nullptr;
    ListNode* head = new ListNode(vals[0]);
    ListNode* curr = head;
    for (size_t i = 1; i < vals.size(); ++i) {
        curr->next = new ListNode(vals[i]);
        curr = curr->next;
    }
    return head;
}

// 打印链表
void printList(ListNode* head) {
    while (head) {
        std::cout << head->val;
        if (head->next) std::cout << " -> ";
        head = head->next;
    }
    std::cout << std::endl;
}

// 【修复】释放链表内存
void deleteList(ListNode* head) {
    while (head) {
        ListNode* temp = head;
        head = head->next;
        delete temp;
    }
}

// 验证链表是否升序
bool isSorted(ListNode* head) {
    if (!head) return true;
    while (head->next) {
        if (head->val > head->next->val) return false;
        head = head->next;
    }
    return true;
}

// ==================== Main ====================

void testCase(const std::string& name, const std::vector<int>& data) {
    std::cout << "\n【测试】" << name << std::endl;
    ListNode* head = createList(data);
    
    std::cout << "排序前: ";
    printList(head);
    
    Solution sol;
    head = sol.sortList(head);
    
    std::cout << "排序后: ";
    printList(head);
    std::cout << "验证: " << (isSorted(head) ? "✓ 升序" : "✗ 错误") << std::endl;
    
    deleteList(head);  // 释放内存
}

int main() {
    // 原测试用例
    testCase("随机数组", {4, 2, 1, 3, 5, -1, 0});
    
    // 边界测试
    testCase("空链表", {});
    testCase("单节点", {42});
    testCase("已升序", {1, 2, 3, 4, 5});
    testCase("已降序", {5, 4, 3, 2, 1});
    testCase("重复元素", {3, 1, 4, 1, 5, 9, 2, 6, 5});
    testCase("两个节点", {2, 1});
    testCase("全相同", {7, 7, 7, 7});
    
    return 0;
}
```
