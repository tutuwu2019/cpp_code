### LRU cache 
设计和实现一个LRU 缓存机制，支持 get 和 put 操作，要求时间复杂度为o(1)

```cpp
#include <iostream>
#include <unordered_map>
#include <list>

class LRUCache{
private:
  int capacity;
  std::list<std::pair<int, int>> cache;
  std::unordered_map<int, std::list<std::pair<int, int>>::interator> map;

public:
  LRUCache(int cap) : capacity(cap){};

  int get(int key){
    auto it = map.find(key);
    if(it == map::end()){
      return -1;
    }

    cache.splice(cache.begin(), cache, it->second);
    return it->second->second;
  }

  void put(int key, int value){
    auto it = map.find(key);

    if(it != map.end()){
      it->second->second = value;
      cache.splice(cache.begin(), cache, it->second);
      return;
    }

    if(cache.size() == capacity){
      int lastKey = cache.back().first;
      map.erase(lastKey);
      cache.pop_back();
    }

    cache.push_front({ley, value});
    map[key] = cache.begin();
  }

};

int main() {
    LRUCache lru(2); // 容量为 2
    lru.put(1, 1);
    lru.put(2, 2);
    std::cout << lru.get(1) << std::endl;    // 返回 1
    lru.put(3, 3);                         // 导致 key 2 被逐出
    std::cout << lru.get(2) << std::endl;    // 返回 -1 (未找到)
    lru.put(4, 4);                         // 导致 key 1 被逐出
    std::cout << lru.get(1) << std::endl;    // 返回 -1
    std::cout << lru.get(3) << std::endl;    // 返回 3
    std::cout << lru.get(4) << std::endl;    // 返回 4
    return 0;
}
```

### 合并 k 个升序链表

给你一个链表数组，每个链表都已经按照升序排列，请你将所有链表合并到一个升序链表中返回。优先使用队列 (堆) 的使用或者分治的思想，时间复杂度 o (nlogk)

```cpp
#include <iostream>
#include <vector>
#include <queue>

// 链表节点定义
struct ListNode {
    int val;
    ListNode *next;
    ListNode() : val(0), next(nullptr) {}
    ListNode(int x) : val(x), next(nullptr) {}
    ListNode(int x, ListNode *next) : val(x), next(next) {}
};

class Solution {
public:
    // 自定义比较函数：val 越小的优先级越高（放在堆顶）
    struct compare {
        bool operator()(ListNode* a, ListNode* b) {
            return a->val > b->val; 
        }
    };

    ListNode* mergeKLists(std::vector<ListNode*>& lists) {
        // 定义最小堆
        std::priority_queue<ListNode*, std::vector<ListNode*>, compare> pq;

        // 1. 将所有非空链表的头节点放入堆中
        for (auto head : lists) {
            if (head) pq.push(head);
        }

        // 2. 创建虚拟头节点，简化链表拼接
        ListNode dummy(0);
        ListNode* tail = &dummy;

        // 3. 循环取出堆顶元素（当前最小值）
        while (!pq.empty()) {
            ListNode* node = pq.top();
            pq.pop();

            tail->next = node;  // 接到结果链表后
            tail = tail->next;

            // 如果该节点还有下一个节点，将其压入堆中
            if (node->next) {
                pq.push(node->next);
            }
        }

        return dummy.next;
    }
};

// ==================== 测试辅助函数 ====================

// 根据 vector 创建链表
ListNode* createList(const std::vector<int>& nums) {
    if (nums.empty()) return nullptr;
    ListNode* head = new ListNode(nums[0]);
    ListNode* cur = head;
    for (size_t i = 1; i < nums.size(); i++) {
        cur->next = new ListNode(nums[i]);
        cur = cur->next;
    }
    return head;
}

// 打印链表
void printList(ListNode* head) {
    if (!head) {
        std::cout << "NULL" << std::endl;
        return;
    }
    while (head) {
        std::cout << head->val;
        if (head->next) std::cout << " -> ";
        head = head->next;
    }
    std::cout << std::endl;
}

// 释放链表内存
void deleteList(ListNode* head) {
    while (head) {
        ListNode* temp = head;
        head = head->next;
        delete temp;
    }
}

// ==================== Main 函数 ====================

int main() {
    Solution sol;
    
    std::cout << "========== 测试 1: 正常情况 ==========" << std::endl;
    {
        // lists = [[1,4,5],[1,3,4],[2,6]]
        std::vector<ListNode*> lists;

        lists.push_back(createList({1, 3, 4}));

            std::cout << "  List " << i << ": ";

        }
        
        ListNode* result = sol.mergeKLists(lists);
        std::cout << "合并结果: ";

    }

        printList(result);  // 期望: NULL
    }
```

分治思想
```cpp
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
        if (!l1) return l2;
        if (!l2) return l1;
        if (l1->val < l2->val) {
            l1->next = mergeTwoLists(l1->next, l2);
            return l1;
        } else {
            l2->next = mergeTwoLists(l1, l2->next);
            return l2;
        }
    }

    ListNode* solve(std::vector<ListNode*>& lists, int left, int right) {
        if (left == right) return lists[left];
        if (left > right) return nullptr;
        int mid = left + (right - left) / 2;
        return mergeTwoLists(solve(lists, left, mid), solve(lists, mid + 1, right));
    }

    ListNode* mergeKLists(std::vector<ListNode*>& lists) {
        return solve(lists, 0, lists.size() - 1);
    }
};
```

### 链表中 k 个元素翻转

```cpp
#include <iostream>

struct ListNode {
    int val;
    ListNode *next;
    ListNode(int x) : val(x), next(nullptr) {}
};

class Solution {
public:
    /**
     * 翻转区间 [a, b) 内的节点（左闭右开）
     */
    ListNode* reverse(ListNode* a, ListNode* b) {
        ListNode* prev = nullptr;
        ListNode* curr = a;
        ListNode* nxt = a;
        while (curr != b) {
            nxt = curr->next;
            curr->next = prev;
            prev = curr;
            curr = nxt;
        }
        // 返回翻转后的新头节点
        return prev;
    }

    ListNode* reverseKGroup(ListNode* head, int k) {
        if (head == nullptr) return nullptr;
        
        // 1. 寻找待翻转区间的终点 b (第 k+1 个节点)
        ListNode* a = head;
        ListNode* b = head;
        for (int i = 0; i < k; i++) {
            // 不足 k 个节点，无需翻转，直接返回
            if (b == nullptr) return head;
            b = b->next;
        }

        // 2. 翻转前 k 个节点
        ListNode* newHead = reverse(a, b);

        // 3. 递归翻转后续链表，并连接起来
        // 此时 a 已经变成了翻转后这组的尾节点
        a->next = reverseKGroup(b, k);

        return newHead;
    }
};

// 辅助函数：打印链表
void printList(ListNode* head) {
    while (head) {
        std::cout << head->val << (head->next ? " -> " : "");
        head = head->next;
    }
    std::cout << std::endl;
}
```

### 排序链表

给定链表的头节点，请将其按照升序排列并返回排序后的链表，要求时间复杂度 o (nlogn)，空间复杂度 o (1)，并写一个 main 测试

```cpp
#include <iostream>
#include <vector>

struct ListNode {
    int val;
    ListNode *next;
    ListNode(int x) : val(x), next(nullptr) {}
};

class Solution {
public:
    // 合并两个升序链表
    ListNode* merge(ListNode* l1, ListNode* l2) {
        ListNode dummy(0);
        ListNode* curr = &dummy;
        while (l1 && l2) {
            if (l1->val < l2->val) {
                curr->next = l1;
                l1 = l1->next;
            } else {
                curr->next = l2;
                l2 = l2->next;
            }
            curr = curr->next;
        }
        curr->next = l1 ? l1 : l2;
        return dummy.next;
    }

    // 自底向上归并排序
    ListNode* sortList(ListNode* head) {
        if (!head || !head->next) return head;

        // 获取链表长度
        int length = 0;
        ListNode* node = head;
        while (node) {
            length++;
            node = node->next;
        }

        ListNode dummy(0);
        dummy.next = head;

        // 自底向上：从 subLen=1 开始，每次翻倍
        for (int subLen = 1; subLen < length; subLen <<= 1) {
            ListNode* prev = &dummy;
            ListNode* curr = dummy.next;

            while (curr) {
                // 找到第一部分 [head1, ...]，长度为 subLen
                ListNode* head1 = curr;
                for (int i = 1; i < subLen && curr->next; ++i) {
                    curr = curr->next;
                }
                
                // 切断第一部分，保存第二部分起点
                ListNode* head2 = curr->next;
                curr->next = nullptr;
                curr = head2;

                // 找到第二部分，长度最多为 subLen
                for (int i = 1; i < subLen && curr && curr->next; ++i) {
                    curr = curr->next;
                }

                // 保存下一组起点，切断第二部分
                ListNode* nextGroup = nullptr;
                if (curr) {
                    nextGroup = curr->next;
                    curr->next = nullptr;
                }

                // 合并当前两部分，并接到 prev 后面
                prev->next = merge(head1, head2);
                
                // 移动 prev 到当前合并后链表的末尾
                while (prev->next) {
                    prev = prev->next;
                }
                
                // 处理下一组
                curr = nextGroup;
            }
        }
        return dummy.next;
    }
};

// ==================== 测试辅助函数 ====================

    if (vals.empty()) return nullptr;

        head = head->next;
    }

}

// ==================== Main ====================
```

### 环形链表 II + 相交链表

给定一个链表，返回链表开始入环的第一个节点；找到两个单链表相交的起始节点。
> 快慢指针(Floyd 判圈)、数学推导(为什么相遇后从头再走能到入口)

```cpp
#include <iostream>
#include <vector>
#include <unordered_set>

struct ListNode {
    int val;
    ListNode *next;
    ListNode(int x) : val(x), next(nullptr) {}
};

class Solution {
public:
    // ==================== 142. 环形链表 II ====================
    ListNode *detectCycle(ListNode *head) {
        if (!head || !head->next) return nullptr;

        ListNode *slow = head, *fast = head;
        bool met = false;  // 修复一：用标志位明确区分两种退出路径

        while (fast && fast->next) {
            slow = slow->next;
            fast = fast->next->next;
            if (slow == fast) {
                met = true;
                break;
            }
        }

        if (!met) return nullptr;  // 自然退出 = 无环

        slow = head;
        while (slow != fast) {
            slow = slow->next;
            fast = fast->next;
        }
        return slow;
    }

    // 修复三：不修改入参，改用局部变量
    ListNode *detectCycleHash(ListNode *head) {
        std::unordered_set<ListNode*> visited;
        ListNode *curr = head;  // 使用局部指针
        while (curr) {
            if (visited.count(curr)) return curr;
            visited.insert(curr);
            curr = curr->next;
        }
        return nullptr;
    }

    // ==================== 160. 相交链表 ====================
    ListNode *getIntersectionNode(ListNode *headA, ListNode *headB) {
        if (!headA || !headB) return nullptr;

        ListNode *pA = headA, *pB = headB;
        while (pA != pB) {
            pA = pA ? pA->next : headB;
            pB = pB ? pB->next : headA;
        }
        return pA;
    }

    ListNode *getIntersectionNodeByLen(ListNode *headA, ListNode *headB) {
        int lenA = 0, lenB = 0;
        ListNode *currA = headA, *currB = headB;

        while (currA) { lenA++; currA = currA->next; }
        while (currB) { lenB++; currB = currB->next; }

        currA = headA; currB = headB;
        while (lenA > lenB) { currA = currA->next; lenA--; }
        while (lenB > lenA) { currB = currB->next; lenB--; }

        while (currA != currB) {
            currA = currA->next;
            currB = currB->next;
        }
        return currA;
    }
};

// ==================== 测试辅助函数 ====================

    ListNode* head = new ListNode(vals[0]);

        ListNode* temp = head;

    }
}
```

### 二叉树的序列化与反序列化
请设计一个算法来实现二叉树的序列化和反序列化，保证任何二叉树都能被序列化为一个字符串，并且通过这个字符串能够还原二叉树结构

> DFS(前序/后序)与BFS 的灵活运用，空节点标记处理，递归/迭代实现能力

```cpp
#include <iostream>
#include <string>
#include <vector>
#include <queue>
#include <sstream>
#include <cassert>

using namespace std;

struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
};

// ==================== 解法 1: 前序遍历 (DFS) ====================
// 格式: 根,左,右  (空节点用 "#" 表示)
class CodecDFS {
public:
    // 序列化：前序遍历
    string serialize(TreeNode* root) {
        if (!root) return "#";
        return to_string(root->val) + "," + serialize(root->left) + "," + serialize(root->right);
    }

    // 反序列化
    TreeNode* deserialize(string data) {
        vector<string> nodes;
        stringstream ss(data);
        string item;
        
        // 按逗号分割字符串
        while (getline(ss, item, ',')) {
            nodes.push_back(item);
        }
        
        int index = 0;
        return buildTree(nodes, index);
    }

private:
    TreeNode* buildTree(const vector<string>& nodes, int& index) {
        if (index >= nodes.size() || nodes[index] == "#") {
            index++;
            return nullptr;
        }
        
        TreeNode* node = new TreeNode(stoi(nodes[index]));
        index++;
        node->left = buildTree(nodes, index);
        node->right = buildTree(nodes, index);
        return node;
    }
};

// ==================== 解法 2: 层序遍历 (BFS) ====================
// 格式: 按层遍历，空节点用 "#" 占位，保证完全二叉树结构
class CodecBFS {
public:
    // 序列化
    string serialize(TreeNode* root) {
        if (!root) return "#";
        
        string result;
        queue<TreeNode*> q;
        q.push(root);
        
        while (!q.empty()) {
            TreeNode* node = q.front();
            q.pop();
            
            if (node) {
                result += to_string(node->val) + ",";
                q.push(node->left);   // 空指针也入队
                q.push(node->right);
            } else {
                result += "#,";
            }
        }
        
        // 去除末尾多余的 "#,"（可选优化）
        while (result.size() > 2 && result.substr(result.size()-2) == "#,") {
            result.pop_back();
            result.pop_back();
        }
        
        return result.empty() ? "#" : result.substr(0, result.size()-1);
    }

    // 反序列化
    TreeNode* deserialize(string data) {
        if (data == "#") return nullptr;
        
        vector<string> nodes = split(data, ',');
        TreeNode* root = new TreeNode(stoi(nodes[0]));
        queue<TreeNode*> q;
        q.push(root);
        
        int index = 1;
        while (!q.empty() && index < nodes.size()) {
            TreeNode* parent = q.front();
            q.pop();
            
            // 左孩子
            if (index < nodes.size() && nodes[index] != "#") {
                parent->left = new TreeNode(stoi(nodes[index]));
                q.push(parent->left);
            }
            index++;
            
            // 右孩子
            if (index < nodes.size() && nodes[index] != "#") {
                parent->right = new TreeNode(stoi(nodes[index]));
                q.push(parent->right);
            }
            index++;
        }
        
        return root;
    }

private:
    vector<string> split(const string& s, char delimiter) {
        vector<string> tokens;
        string token;
        stringstream ss(s);
        while (getline(ss, token, delimiter)) {
            tokens.push_back(token);
        }
        return tokens;
    }
};
```

### 二叉树的最近公共祖先(LCA)
给定二叉树的根节点和两个节点p、q，找到它们的最近公共祖先。

> 递归思想(后序遍历)，分治策略。

```cpp
#include <iostream>
#include <unordered_map>
#include <unordered_set>
#include <vector>
#include <queue>

using namespace std;

struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
};

class Solution {
public:
    // ==================== 解法 1: 递归 DFS (最优) ====================
    // 时间: O(N), 空间: O(H) - H 为树高度
    TreeNode* lowestCommonAncestor(TreeNode* root, TreeNode* p, TreeNode* q) {
        // base case: 空节点，或找到 p/q
        if (!root || root == p || root == q) {
            return root;
        }
        
        // 在左右子树中递归查找
        TreeNode* left = lowestCommonAncestor(root->left, p, q);
        TreeNode* right = lowestCommonAncestor(root->right, p, q);
        
        // 情况 1: p 和 q 分别在左右子树 -> 当前节点是 LCA
        if (left && right) {
            return root;
        }
        
        // 情况 2: 都在左子树 -> left 是 LCA
        // 情况 3: 都在右子树 -> right 是 LCA
        // 情况 4: 都没找到 -> 返回 nullptr
        return left ? left : right;
    }

    // ==================== 解法 2: 存储父节点 (适用于多次查询) ====================
    // 预处理: O(N) 时间, O(N) 空间
    // 每次查询: O(H) 时间, O(H) 空间
    void preprocess(TreeNode* root) {
        parentMap[root] = nullptr;
        buildParentMap(root);
    }
    
    TreeNode* lowestCommonAncestorV2(TreeNode* p, TreeNode* q) {
        unordered_set<TreeNode*> ancestors;
        
        // 将 p 的所有祖先加入集合
        while (p) {
            ancestors.insert(p);
            p = parentMap[p];
        }
        
        // 找 q 的祖先中第一个在集合中的
        while (q) {
            if (ancestors.count(q)) {
                return q;
            }
            q = parentMap[q];
        }
        return nullptr;
    }

private:
    unordered_map<TreeNode*, TreeNode*> parentMap;
    
    void buildParentMap(TreeNode* node) {
        if (!node) return;
        
        if (node->left) {
            parentMap[node->left] = node;
            buildParentMap(node->left);
        }
        if (node->right) {
            parentMap[node->right] = node;
            buildParentMap(node->right);
        }
    }
};

class SolutionBST {
public:
    TreeNode* lowestCommonAncestor(TreeNode* root, TreeNode* p, TreeNode* q) {
        // 确保 p < q
        if (p->val > q->val) swap(p, q);
        
        while (root) {
            if (root->val > q->val) {
                // p,q 都在左子树
                root = root->left;
            } else if (root->val < p->val) {
                // p,q 都在右子树
                root = root->right;
            } else {
                // 当前节点在 [p, q] 区间内，就是 LCA
                return root;
            }
        }
        return nullptr;
    }
};
```

### 二叉搜索树与双向链表
输入一个二叉搜索树，将该BST 专程一个排序的双向链表，要求不能常见任何新的节点。

> BST 中序遍历特性，递归中的全局变量/引用传递，树结构指针的重定向


```cpp
#include <iostream>
#include <vector>
#include <stack>

using namespace std;

struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
};

class Solution {
public:
    // ==================== 解法 1: 递归中序遍历 ====================
    TreeNode* Convert(TreeNode* root) {
        if (!root) return nullptr;

        TreeNode* prev = nullptr;
        TreeNode* head = nullptr;

        inorder(root, prev, head);
        return head;
    }

    void inorder(TreeNode* node, TreeNode*& prev, TreeNode*& head) {
        if (!node) return;

        inorder(node->left, prev, head);

        if (prev) {
            prev->right = node;
            node->left = prev;
        } else {
            head = node;
        }
        prev = node;

        inorder(node->right, prev, head);
    }

    // ==================== 解法 2: 迭代中序遍历 ====================
    TreeNode* ConvertIterative(TreeNode* root) {
        if (!root) return nullptr;

        stack<TreeNode*> stk;
        TreeNode* curr = root;
        TreeNode* prev = nullptr;
        TreeNode* head = nullptr;

        while (curr || !stk.empty()) {
            while (curr) {
                stk.push(curr);
                curr = curr->left;
            }

            curr = stk.top();
            stk.pop();

            if (prev) {
                prev->right = curr;
                curr->left = prev;
            } else {
                head = curr;
            }
            prev = curr;

            curr = curr->right;
        }

        return head;
    }
};

// ==================== 测试辅助函数 ====================

TreeNode* createBST() {
    TreeNode* root = new TreeNode(10);
    root->left = new TreeNode(6);
    root->right = new TreeNode(14);
    root->left->left = new TreeNode(4);
    root->left->right = new TreeNode(8);
    root->right->left = new TreeNode(12);
    root->right->right = new TreeNode(16);
    return root;
}

void printForward(TreeNode* head) {
    cout << "正向遍历: ";
    TreeNode* tail = nullptr;
    while (head) {
        cout << head->val;
        if (head->right) cout << " <-> ";
        tail = head;
        head = head->right;
    }
    cout << endl;

    cout << "反向遍历: ";
    while (tail) {
        cout << tail->val;
        if (tail->left) cout << " <-> ";
        tail = tail->left;
    }
    cout << endl;
}

bool verifySorted(TreeNode* head) {
    if (!head) return true;
    while (head->right) {
        if (head->val >= head->right->val) return false;
        head = head->right;
    }
    return true;
}

void deleteList(TreeNode* head) {
    while (head) {
        TreeNode* temp = head;
        head = head->right;
        delete temp;
    }
}

// ==================== Main ====================

    delete list3;
```

### 二叉树中的最大路径和

路径被定义为一条从树中任意节点触发，沿父节点-子节点连接，达到任意节点的序列。求路径和的最大值

> 递归返回值的设计(但路径最大和vs 全局路径和)，负值处理，全局状态维护


```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>

using namespace std;

struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
};

class Solution {
public:
    int maxPathSum(TreeNode* root) {
        maxSum = INT_MIN;
        maxGain(root);
        return maxSum;
    }

private:
    int maxSum;  // 全局最大路径和
    
    // 返回：以当前节点为端点的最大路径和（向上贡献给父节点的值）
    int maxGain(TreeNode* node) {
        if (!node) return 0;
        
        // 递归计算左右子树的最大贡献值
        // max(0, ...) 表示如果贡献值为负，则不选该子树（路径不延伸）
        int leftGain = max(maxGain(node->left), 0);
        int rightGain = max(maxGain(node->right), 0);
        
        // 以当前节点为"最高点"的路径和（左-根-右）
        // 这是一个候选答案，因为路径可以在此转折
        int priceNewPath = node->val + leftGain + rightGain;
        
        // 更新全局最大值
        maxSum = max(maxSum, priceNewPath);
        
        // 返回给父节点的值：只能选择左或右中的一边（因为路径不能分叉）
        // 相当于以当前节点为端点，向上延伸的最大路径
        return node->val + max(leftGain, rightGain);
    }
};

// ==================== 测试辅助函数 ====================

TreeNode* createTree(const vector<int>& vals, int index = 0) {
    if (index >= vals.size() || vals[index] == INT_MIN) return nullptr;
    TreeNode* root = new TreeNode(vals[index]);
    root->left = createTree(vals, 2 * index + 1);
    root->right = createTree(vals, 2 * index + 2);
    return root;
}

void deleteTree(TreeNode* root) {
    if (!root) return;
    deleteTree(root->left);
    deleteTree(root->right);
    delete root;
}

// ==================== 测试用例 ====================

    Solution sol;
    
    cout << "========== 二叉树最大路径和测试 ==========" << endl;
    
    // 测试 1: 题目示例
    //        1
    //       / \
    //      2   3
    // 最大路径: 2 -> 1 -> 3 = 6
    TreeNode* root1 = new TreeNode(1);
    root1->left = new TreeNode(2);
    root1->right = new TreeNode(3);
    cout << "测试 1 (简单树): " << sol.maxPathSum(root1) << " (期望: 6)" << endl;
    deleteTree(root1);
    
    // 测试 2: 包含负数
    //        -10
    //        / \
    //       9  20
    //         /  \
    //        15   7
    // 最大路径: 15 -> 20 -> 7 = 42
    TreeNode* root2 = new TreeNode(-10);
    root2->left = new TreeNode(9);
    root2->right = new TreeNode(20);
    root2->right->left = new TreeNode(15);

    //        -3
    //       / \
    //     -2   -1
```

### 将有序数组转换为高度平衡二叉搜索树

给你一个整数数组 nums，其中元素已经按升序排列，请你将其转换为一课高度平衡二叉搜索树。
```cpp
#include <iostream>
#include <vector>
#include <cmath>

using namespace std;

struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
};

class Solution {
public:
    TreeNode* sortedArrayToBST(vector<int>& nums) {
        return build(nums, 0, nums.size() - 1);
    }
    
private:
    TreeNode* build(const vector<int>& nums, int left, int right) {
        // 递归终止条件
        if (left > right) return nullptr;
        
        // 选择中间位置元素作为根节点
        // 使用 left + (right - left) / 2 防止溢出
        int mid = left + (right - left) / 2;
        
        TreeNode* root = new TreeNode(nums[mid]);
        
        // 递归构建左右子树
        root->left = build(nums, left, mid - 1);
        root->right = build(nums, mid + 1, right);
        
        return root;
    }
};
```

### 验证二叉搜索树
给定一个二叉树的根节点，判断其是否是一个有效的二叉搜索树

> BST 定义的深刻理解(左子树所有节点 < 根  < 有子树所有节点) 中序遍历验证，递归上下界传递
>

```cpp

#include <iostream>
#include <vector>
#include <stack>
#include <climits>

using namespace std;

struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
};

class Solution {
public:
    // ==================== 解法 1: 递归传递范围 (最优) ====================
    // 时间: O(n), 空间: O(h)
    bool isValidBST(TreeNode* root) {
        return validate(root, nullptr, nullptr);
    }
    
    // 使用 long long 防止 INT_MIN/INT_MAX 边界问题
    bool validate(TreeNode* node, TreeNode* minNode, TreeNode* maxNode) {
        if (!node) return true;
        
        // 当前节点必须满足: minNode->val < node->val < maxNode->val
        if (minNode && node->val <= minNode->val) return false;
        if (maxNode && node->val >= maxNode->val) return false;
        
        // 左子树更新上界为当前节点值，右子树更新下界为当前节点值
        return validate(node->left, minNode, node) && 
               validate(node->right, node, maxNode);
    }

    // ==================== 解法 2: 中序遍历 (利用 BST 性质) ====================
    // BST 的中序遍历必须是严格升序
    // 时间: O(n), 空间: O(h)
    bool isValidBSTInorder(TreeNode* root) {
        TreeNode* prev = nullptr;
        return inorder(root, prev);
    }
    
    bool inorder(TreeNode* node, TreeNode*& prev) {
        if (!node) return true;
        
        // 遍历左子树
        if (!inorder(node->left, prev)) return false;
        
        // 检查当前节点（中序位置）
        if (prev && node->val <= prev->val) return false;
        prev = node;
        
        // 遍历右子树
        return inorder(node->right, prev);
    }

    // ==================== 解法 3: 迭代中序遍历 ====================
    // 时间: O(n), 空间: O(h)
    bool isValidBSTIterative(TreeNode* root) {
        if (!root) return true;
        
        stack<TreeNode*> stk;
        TreeNode* curr = root;
        TreeNode* prev = nullptr;
        
        while (curr || !stk.empty()) {
            while (curr) {
                stk.push(curr);
                curr = curr->left;
            }
            
            curr = stk.top();
            stk.pop();
            
            // 检查升序
            if (prev && curr->val <= prev->val) return false;
            prev = curr;
            
            curr = curr->right;
        }
        return true;
    }
};

// ==================== 测试辅助函数 ====================

TreeNode* createTree(const vector<int>& vals, int nullVal = INT_MIN) {
    if (vals.empty()) return nullptr;
    vector<TreeNode*> nodes;
    for (int v : vals) {
        nodes.push_back(v == nullVal ? nullptr : new TreeNode(v));
    }
    for (size_t i = 0; i < nodes.size(); ++i) {
        if (nodes[i]) {
            if (2*i+1 < nodes.size()) nodes[i]->left = nodes[2*i+1];
            if (2*i+2 < nodes.size()) nodes[i]->right = nodes[2*i+2];
        }
    }
    return nodes[0];
}

void deleteTree(TreeNode* root) {
    if (!root) return;
    deleteTree(root->left);
    deleteTree(root->right);
    delete root;
}

// ==================== 测试用例 ====================
```

### 课程表(拓扑排序)

你这个学习必须选修 numCourses 门课程，记为0到 numCourses-1。先修课程按数组 prerequisites 给出，判断你是否能完成所有课程的学习。

> 有向图环检测，拓扑排序(Kahn 算法BFS 或者DFS 三色标级)，入度、邻接表建图

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <cstring>

using namespace std;

class Solution {
public:
    // ==================== 解法 1: Kahn算法 (BFS 拓扑排序) ====================
    // 时间: O(V + E), 空间: O(V + E)
    bool canFinishBFS(int numCourses, vector<vector<int>>& prerequisites) {
        // 1. 构建图和入度数组
        vector<vector<int>> graph(numCourses);  // 邻接表
        vector<int> inDegree(numCourses, 0);     // 入度数组
        
        for (auto& pre : prerequisites) {
            int course = pre[0];
            int prerequisite = pre[1];
            graph[prerequisite].push_back(course);  // 先修课指向当前课
            inDegree[course]++;                      // 当前课入度+1
        }
        
        // 2. 将所有入度为0的课程入队（可以直接学的课）
        queue<int> q;
        for (int i = 0; i < numCourses; ++i) {
            if (inDegree[i] == 0) q.push(i);
        }
        
        // 3. 拓扑排序
        int visited = 0;  // 记录已修课程数
        while (!q.empty()) {
            int curr = q.front();
            q.pop();
            visited++;
            
            // 修完 curr，减少后续课程的入度
            for (int next : graph[curr]) {
                inDegree[next]--;
                if (inDegree[next] == 0) {
                    q.push(next);  // 先修课都修完了，可以学了
                }
            }
        }
        
        // 如果所有课程都被修完，则无环
        return visited == numCourses;
    }

    // ==================== 解法 2: DFS 检测环 ====================
    // 时间: O(V + E), 空间: O(V + E)
    bool canFinishDFS(int numCourses, vector<vector<int>>& prerequisites) {
        vector<vector<int>> graph(numCourses);
        for (auto& pre : prerequisites) {
            graph[pre[1]].push_back(pre[0]);  // 先修课 -> 当前课
        }
        
        // 状态标记: 0=未访问, 1=访问中(在递归栈中), 2=已访问
        vector<int> state(numCourses, 0);
        
        for (int i = 0; i < numCourses; ++i) {
            if (state[i] == 0) {
                if (hasCycle(graph, state, i)) return false;
            }
        }
        return true;
    }
    
private:
    bool hasCycle(vector<vector<int>>& graph, vector<int>& state, int curr) {
        if (state[curr] == 1) return true;   // 遇到访问中的节点，有环
        if (state[curr] == 2) return false;  // 已处理过，无环
        
        state[curr] = 1;  // 标记为访问中
        
        for (int next : graph[curr]) {
            if (hasCycle(graph, state, next)) return true;
        }
        
        state[curr] = 2;  // 标记为已访问
        return false;
    }
};

// ==================== 测试 ====================

void testCourseSchedule() {
    Solution sol;
    
    cout << "========== 课程表 (拓扑排序) 测试 ==========" << endl;
    
    // 测试 1: 无环，可以完成
    // 课程 0 需要课程 1 作为先修
    // 课程 1 需要课程 2 作为先修
    // 路径: 2 -> 1 -> 0
    vector<vector<int>> pre1 = {{0, 1}, {1, 2}};
    cout << "测试 1 (线性依赖): " 
         << (sol.canFinishBFS(3, pre1) ? "可以完成" : "无法完成") 
         << " (期望: 可以完成)" << endl;
    
    // 测试 2: 有环，无法完成
    // 0 -> 1 -> 0 (循环依赖)
    vector<vector<int>> pre2 = {{0, 1}, {1, 0}};
    cout << "测试 2 (循环依赖): " 
         << (sol.canFinishBFS(2, pre2) ? "可以完成" : "无法完成")
         << " (期望: 无法完成)" << endl;
    
    // 测试 3: 复杂无环
    // 课程: 0,1,2,3,4
    // 依赖: 1->0, 2->0, 3->1, 3->2, 4->3
    vector<vector<int>> pre3 = {{0,1}, {0,2}, {1,3}, {2,3}, {3,4}};
    cout << "测试 3 (复杂DAG): " 
         << (sol.canFinishDFS(5, pre3) ? "可以完成" : "无法完成")
         << " (期望: 可以完成)" << endl;
    
    // 测试 4: 复杂有环
    // 0->1->2->0 环，同时 3->4
    vector<vector<int>> pre4 = {{1,0}, {2,1}, {0,2}, {4,3}};
    cout << "测试 4 (部分有环): " 
         << (sol.canFinishDFS(5, pre4) ? "可以完成" : "无法完成")
         << " (期望: 无法完成)" << endl;
    
    // 测试 5: 无先修课程
    vector<vector<int>> pre5;
    cout << "测试 5 (无依赖): " 
         << (sol.canFinishBFS(3, pre5) ? "可以完成" : "无法完成")
         << " (期望: 可以完成)" << endl;
    
    // 测试 6: 单课程自环
```

### 网络延迟时间(单源最短路径)
有n 个网络节点，标级为1到n ，给定一个列表 times 表示信号经过有向边的传递时间。从某个节点k 发出信号，求所有节点收到信号的最短时间，若无法到达所有节点返回-1

> Dijkstra 算法(堆优化)，松弛操作，负权边判断(是否需要Bellman-Ford)

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <climits>
#include <algorithm>

using namespace std;

class Solution {
public:
    int networkDelayTime(vector<vector<int>>& times, int n, int k) {
        // 1. 构建邻接表: graph[u] = {(v, w), ...} 表示 u->v 权重为w
        vector<vector<pair<int, int>>> graph(n + 1);  // 节点编号1~n
        for (auto& time : times) {
            int u = time[0], v = time[1], w = time[2];
            graph[u].push_back({v, w});
        }
        
        // 2. 初始化距离数组，dist[i] 表示从k到i的最短时间
        vector<int> dist(n + 1, INT_MAX);
        dist[k] = 0;  // 起点到自身距离为0
        
        // 3. 优先队列: {距离, 节点}，按距离从小到大（小顶堆）
        // 使用greater实现小顶堆，默认是大顶堆
        priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
        pq.push({0, k});  // {距离, 节点}
        
        // 4. Dijkstra算法
        while (!pq.empty()) {
            auto [d, u] = pq.top();
            pq.pop();
            
            // 如果当前距离大于已记录的最短距离，跳过（过期数据）
            if (d > dist[u]) continue;
            
            // 松弛操作：更新u的所有邻居
            for (auto& [v, w] : graph[u]) {
                if (dist[u] + w < dist[v]) {
                    dist[v] = dist[u] + w;  // 更新最短距离
                    pq.push({dist[v], v});  // 加入队列
                }
            }
        }
        
        // 5. 找到到达所有节点的最长时间（即最后收到的信号）
        int maxTime = 0;
        for (int i = 1; i <= n; ++i) {
            if (dist[i] == INT_MAX) return -1;  // 有节点不可达
            maxTime = max(maxTime, dist[i]);
        }
        
        return maxTime;
    }
    
    // ==================== Bellman-Ford解法 (适用于负权边，但本题是正权) ====================
    int networkDelayTimeBellman(vector<vector<int>>& times, int n, int k) {
        vector<int> dist(n + 1, INT_MAX);
        dist[k] = 0;
        
        // 松弛 n-1 次
        for (int i = 0; i < n - 1; ++i) {
            bool updated = false;
            for (auto& time : times) {
                int u = time[0], v = time[1], w = time[2];
                if (dist[u] != INT_MAX && dist[u] + w < dist[v]) {
                    dist[v] = dist[u] + w;
                    updated = true;
                }
            }
            if (!updated) break;  // 提前结束
        }
        
        // 检查负权环（本题不需要，因为times都是正数）
        int maxTime = 0;
        for (int i = 1; i <= n; ++i) {
            if (dist[i] == INT_MAX) return -1;
            maxTime = max(maxTime, dist[i]);
        }
        return maxTime;
    }
};

// ==================== 测试 ====================

int main() {
    Solution sol;
    
    cout << "========== 网络延迟时间 (Dijkstra算法) 测试 ==========" << endl;
    
    // 测试 1: 基本案例
    // 图示: 2->1(1), 2->3(1), 3->4(1)
    // 从2出发: 到1需要1, 到3需要1, 到4需要2(经3)
    // 最大时间: 2
    vector<vector<int>> times1 = {{2,1,1}, {2,3,1}, {3,4,1}};
    int result1 = sol.networkDelayTime(times1, 4, 2);
    cout << "测试 1: " << result1 << " (期望: 2)" << endl;

    // 1->2(1), 2->3(2), 1->3(2)
    // 从1出发可以到所有节点，但从2出发到不了1
    vector<vector<int>> times2 = {{1,2,1}, {2,3,2}, {1,3,2}};
    int result2 = sol.networkDelayTime(times2, 3, 2);  // 从2出发
    cout << "测试 2: " << result2 << " (期望: -1, 因为到不了1)" << endl;
    
    // 测试 3: 多条路径选最短的
    // 1->2(1), 2->3(1), 1->3(4)
    // 从1到3: 直接走需要4，经2走需要2，应该选2
    vector<vector<int>> times3 = {{1,2,1}, {2,3,1}, {1,3,4}};
    int result3 = sol.networkDelayTime(times3, 3, 1);
    cout << "测试 3: " << result3 << " (期望: 2)" << endl;
    
    // 测试 4: 单节点
    vector<vector<int>> times4;
    int result4 = sol.networkDelayTime(times4, 1, 1);
    cout << "测试 4: " << result4 << " (期望: 0)" << endl;
    
    // 测试 5: 复杂图
    // 1->2(3), 1->3(1), 3->2(1), 2->4(1), 3->4(4)
    // 从1出发: 到2最短为1->3->2=2, 到3=1, 到4=3(经2)或5(经3)
    vector<vector<int>> times5 = {{1,2,3}, {1,3,1}, {3,2,1}, {2,4,1}, {3,4,4}};
    int result5 = sol.networkDelayTime(times5, 4, 1);
    cout << "测试 5: " << result5 << " (期望: 3)" << endl;
    
    // 测试 Bellman-Ford
    cout << "\n========== Bellman-Ford 验证 ==========" << endl;
    int bf1 = sol.networkDelayTimeBellman(times1, 4, 2);
    cout << "Bellman-Ford 测试1: " << bf1 << " (期望: 2)" << endl;
```
